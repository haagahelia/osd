A PARAMETER

It's the input!

PARAMETER EXAMPLE

(DEFINITION AND CALL)

function decorate(name) {
    console.log("*******************");
    console.log(name);
    console.log("*******************");
}

decorate("John");
decorate("Marie");

RETURN VALUE

It's the output!

RETURN VALUE EXAMPLE

(DEFINITION AND CALL)

function getGreetings(name) {
    return "Hello " + name + "!";
}

// call a function with a parameter and
// save the return value to a variable  
var greetings = getGreetings("Matt");
// you can use the variable value now wherever you want
console.log(greetings);
